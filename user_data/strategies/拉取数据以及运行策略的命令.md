# 下面的命令的作用是运行某个特定类文件
```
docker run --rm -it \
  -v $(pwd)/user_data:/freqtrade/user_data \
  freqtradeorg/freqtrade:stable \
  backtesting \
  --config user_data/config.json \
  --strategy TrendPullbackATR
```

# 拉取特定时间窗口的数据
```
docker run --rm -it \
  -v $(pwd)/user_data:/freqtrade/user_data \
  freqtradeorg/freqtrade:stable \
  download-data \
  --exchange binance \
  --pairs BTC/USDT \
  --timeframes 15m
```

# 拉取2020年到今天的数据
```
docker run --rm -it \
  -v $(pwd)/user_data:/freqtrade/user_data \
  freqtradeorg/freqtrade:stable \
  download-data \
  --exchange binance \
  --pairs BTC/USDT \
  --timeframes 4h \
  --timerange 20200101- \
  --erase
```
